\Def{Pairs / Triples}
    {
       \begin{itemize}
         \item Given two objects x and y, we can produce $(x, y)$.
         \item ...including other pairs: $(x, (y, z))$.
         \item We may write $(x, y, z)$ for notational convenience.
         \item The order of the elements matters.
       \end{itemize}
    }
\Def{Sets}
    {
       \begin{itemize}
         \item Recall sets: $\{a, b, c, ... \}$
         \item Each element may appear in the set only once.
         \item The order of the elements in the set does not matter.
       \end{itemize}
    }
}
