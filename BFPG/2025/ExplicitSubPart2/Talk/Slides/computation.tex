\only<2>{
  \Def{$\beta$-reduction: $(\lambda(x : \alpha).M) \; N
        \rightarrow M[N/x]$}
      {$\beta$-reduction is the essence of computation for the STLC.
      \\ If we see an application after an abstraction, we may 
         perform the given substitution then strip off the lambda
         binder.} 
  \Ex{Example of $beta$-reduction.}
     {\vspace{-0.5cm}
      \begin{align*}
      (\lambda (x : \alpha).x) \; y 
        &\rightarrow \lambda (x : \alpha).x[y/x] 
          &\text{replace abs-app with substitution.}\\
        &\rightarrow \lambda (x : \alpha).y 
          &\text{perform the substitution.}\\
        &\rightarrow y
          &\text{eliminate the relevant binder.}
      \end{align*}
     }
}
