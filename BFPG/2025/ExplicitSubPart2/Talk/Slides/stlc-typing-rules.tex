\Def{STLC: Typing Rules}
    {\[ \trfrac[(\textit{$\rightarrow$-Form})]
             {\Gamma \vdash \alpha \Type \qquad 
              \Gamma \vdash \beta \Type}
            {\Gamma \vdash \alpha \rightarrow \beta \Type}
     \]
     \[
        \trfrac[(\textit{$\rightarrow$-Intro})]
             {\begin{trgather}
                \Gamma \vdash \alpha \rightarrow \beta \Type \qquad
                \Gamma, x_m \ofType \alpha \vdash M \ofType \beta
              \end{trgather}}
             {\Gamma \vdash \lambda(x \ofType \alpha).M \ofType
                \alpha \rightarrow \beta} 
     \]
    }
}

\Def{STLC: Typing Rules}
    {\[ \trfrac[(\textit{$\rightarrow$-Form})]
             {\Gamma \vdash \alpha \Type \qquad 
              \Gamma \vdash \beta \Type}
            {\Gamma \vdash \alpha \rightarrow \beta \Type}
     \]
     \[
        \trfrac[(\textit{$\rightarrow$-Intro})]
             {\begin{trgather}
                \Gamma \vdash \alpha \rightarrow \beta \Type \qquad
                \Gamma, x_m \ofType \alpha \vdash M \ofType \beta
              \end{trgather}}
             {\Gamma \vdash \lambda(x \ofType \alpha).M \ofType
                \alpha \rightarrow \beta} 
     \]
     \[
       \trfrac[(\textit{$\rightarrow$-Elim})]
         {\begin{trgather}
          \Gamma \vdash \alpha \rightarrow \beta \Type \qquad
          \Gamma \vdash M \ofType \alpha \rightarrow \beta \qquad
          \Gamma \vdash N \ofType \alpha
          \end{trgather}}
         {\Gamma \vdash M \; N \ofType \beta}
     \]
      
    }
}
