\only<3>{

  \Def{Rule 2. $y[T/x] = y$ if $x \neq y$}
      {Substituting by a different variable does nothing.}
  \Def{Rule 3. $(M \; N)[T/x] = M[T/x] \; N[T/x]$}
      {Substitution distributes over application.}
}

\only<4>{
  \vspace{-0.7cm}
  \Def{Rule 1. $x[T/x] = T$}
      {When substituting into a variable, we just have the term.}
  \Def{Rule 2. $y[T/x] = y$ if $x \neq y$}
      {Substituting by a different variable does nothing.}
  \Def{Rule 3. $(M \; N)[T/x] = M[T/x] \; N[T/x]$}
      {Substitution distributes over application.}
  \Def{Rule 4. $(\lambda (x : \alpha).M)[T/x] = 
        \lambda (x : \alpha).M$}
      {Abstraction substitution is neutral for the binding 
       variable.}
}

       variable.}
  \Def{Rule 5. $(\lambda (y : \alpha).M)[T/x] = 
         \lambda (y : \alpha).M[T/x]$ if: \\ 
          $x \neq y$ and $y \notin FV(T)$} 
      {Substitution is conditional through an abstraction body.}
}
