\Def{Typing Rules}
    {\[ \trfrac[(\textit{Ty-Arrow})]
             {\Gamma \vdash A \; \Type \qquad 
              \Gamma, x \ofType A \vdash B \; \Type}
            {\Gamma \vdash A \rightarrow B \; \Type}
     \]
     \[
        \trfrac[(\textit{Func})]
             {\begin{trgather}
                \Gamma \vdash A \rightarrow B \Type \qquad
                \Gamma, x \ofType A \vdash M \ofType B
              \end{trgather}}
             {\Gamma \vdash \lambda(x \ofType A).M \ofType
                A \rightarrow B} 
     \]
     
     \[
       \trfrac[(\textit{App})]
         {\begin{trgather}
          \Gamma \vdash A \rightarrow B \Type \qquad
          \Gamma \vdash M \ofType A \rightarrow B \qquad
          \Gamma \vdash N \ofType A
          \end{trgather}}
         {\Gamma \vdash M \; N \ofType B}
     \]
      
    }
}
