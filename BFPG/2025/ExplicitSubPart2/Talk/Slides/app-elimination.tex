\only<1>{
  \textcolor{black}{If the premises hold above, so the conclusion
                    holds below.}
  \Def{The Function Term Eliminator}
      {
     \[
        \boxed{
        \trfrac[(\textit{App})]
             {\qquad \qquad \qquad \qquad}
             {\qquad \qquad \qquad \qquad} 
        }
     \]
     }
}

\only<2>{
  \textcolor{black}{From our program scope $\Gamma$ we can
             obtain \\ some term $M$ with type $A \rightarrow B$.}
  \Def{The Function Term Eliminator}
      {
     \[
        \trfrac[(\textit{App})]
             {
                \boxed{\Gamma \vdash M : A \rightarrow B} 
                      \qquad \qquad \qquad}
             {\qquad \qquad \qquad \qquad} 
     \]
     }
}

\only<3>{
  \textcolor{black}{And, from $\Gamma$ we can also obtain 
                    a term $N$ of type $A$.}
  \Def{The Function Term Eliminator}
      {
     \[
        \trfrac[(\textit{App})]
             { \Gamma \vdash M : A \rightarrow B 
               \qquad \boxed{\Gamma \vdash N : A}}
             {\qquad \qquad \qquad \qquad} 
     \]
     }
}

\only<4>{
  \textcolor{black}{Then we can produce a term $M \; N$  of type $B$.}
  \Def{The Function Term Eliminator}
      {
     \[
        \trfrac[(\textit{App})]
             { \Gamma \vdash M : A \rightarrow B 
                      \qquad \Gamma \vdash N : A}
             {\boxed{\Gamma \vdash M \; N : B }} 
     \]
     }
}
