\only<1>{
\Def{Equational Theory: Category Laws}
    {
     \vspace*{-0.3cm}
     \[ id \circ \gamma = \gamma \qquad \qquad \gamma \circ id = \gamma \]
     \vspace*{-0.3cm}
    }
\Ex{Identity Laws}
    {The identity substitution is both a left and right unit for composition}
}

\only<2>{
\Def{Equational Theory: Category Laws}
    {
     \vspace*{-0.3cm}
     \[ id \circ \gamma = \gamma \qquad \qquad \gamma \circ id = \gamma \]
     \vspace*{-0.3cm}
     \[ (\gamma \circ \delta) \circ \theta = \gamma \circ (\delta \circ \theta) \]
     \vspace*{-0.3cm}
    }
\Ex{Identity and Associativity Laws}
    {Substitution composition forms a category}
}
