\Def{Explicit Substitutions: Subs}
    {

     \begin{align*}

     \end{align*}
     \vspace*{-0.5cm}

\Def{Intuition}
    {
      \begin{itemize}
        \item $id$ is the identity substitution, mapping the
              source context to itself.
        \item $\uparrow$ is the weakening or shifting substituion.
              We 'weaken' the context by one additional variable.
      \end{itemize}
    }
}
