\Def{Untyped Lambda Caluclus}
    {
     \vspace*{-0.5cm}
     \begin{align*}
       V    &::= \; x, y, z, \; ... \\
       M, N &::= \; V 
         &\text{Variable.} \\
            &\;\;\;\;\;\; | \; M\; N  
         &\text{Application.} \\
            &\;\;\;\;\;\; | \; \lambda V.M 
         &\text{Abstraction.}
     \end{align*}
     \vspace*{-0.5cm}
     }
}
