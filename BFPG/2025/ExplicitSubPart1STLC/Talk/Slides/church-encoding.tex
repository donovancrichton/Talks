\Def{Church Naturals}
    {
     \vspace{-0.5cm}
     \begin{align*}
       0 &= \lambda f . \lambda x . x \\
       1 &= \lambda f . \lambda x . f \; x \\
       2 &= \lambda f . \lambda x . f \; (f \; x) \\
       \vdots &
     \end{align*}
     \vspace{-0.5cm}
    }
}
