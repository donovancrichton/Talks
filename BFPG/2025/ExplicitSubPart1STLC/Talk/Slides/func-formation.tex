\only<1>{
  \textcolor{black}{If the premises hold above, so the conclusion
                    holds below.}
  \Def{The Function Term Former}
      {
     \[
        \boxed{
        \trfrac[(\textit{Abs})]
             {\qquad \qquad \qquad \qquad}
             {\qquad \qquad \qquad \qquad} 
        }
     \]
     }
}

\only<2>{
  \textcolor{black}{From our program scope $\Gamma$ we can
             obtain \\ an arbitrary function type $A \rightarrow B$.}
  \Def{The Function Term Former}
      {
     \[
        \trfrac[(\textit{Abs})]
             {
                \boxed{\Gamma \vdash A \rightarrow B \Type} 
                      \qquad \qquad}
             {\qquad \qquad \qquad } 
     \]
     }
}

\only<3>{
  \textcolor{black}{..And..}
  \Def{The Function Term Former}
      {
     \[
        \trfrac[(\textit{Abs})]
             {\Gamma \vdash A \rightarrow B \Type \boxed{\qquad}
              %\Gamma, x \ofType A \vdash M \ofType B
             }
             {\qquad \qquad \qquad} 
     \]
     }
}

\only<4>{
  \textcolor{black}{From our program scope $\Gamma, x : A$ we 
                    can derive a term $M : B$.}
  \Def{The Function Term Former}
      {
     \[
        \trfrac[(\textit{Abs})]
             {\Gamma \vdash A \rightarrow B \Type \qquad
              \boxed{\Gamma, x \ofType A \vdash M \ofType B}
             }
             {\qquad \qquad \qquad} 
     \]
     }
}

\only<5>{
  \textcolor{black}{Then we may derive a lambda abstraction term, 
                    where $x$ has type $A$ and the body is $M$,
                    where the entire abstraction has type
                    $A \rightarrow B$.}
  \Def{The Function Term Former}
      {
     \[
        \trfrac[(\textit{Abs})]
             {\Gamma \vdash A \rightarrow B \Type \qquad
              \Gamma, x \ofType A \vdash M \ofType B
             }
             {\boxed{\Gamma \vdash \lambda(x \ofType A).M \ofType
                A \rightarrow B}}
     \]
     }
}
