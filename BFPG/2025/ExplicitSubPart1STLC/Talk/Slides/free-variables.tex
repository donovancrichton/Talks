
\only<1>{
\Def{FV}
    {A variable is $x$ considered \textit{free} in a lambda term if
     it refers to a binding outside the \textit{scope} of the 
     current lambda term. Defined inductively. }
}

\only<2>{
\Def{The set $FV$ of free variables.}
    {A variable is $x$ considered \textit{free} in a lambda term if
     it refers to a binding outside the \textit{scope} of the 
     current lambda term. Defined inductively. \\
       \vspace{-0.5cm}
       \begin{align*}
         FV(V) &= V \\
         FV(M \; N) &= FV(M) \cup FV(N) \\
         FV(\lambda(V : \alpha).M) &= FV(M) \backslash V 
       \end{align*}
       \vspace{-0.5cm}
   }
}

\only<3>{
\vspace{-0.5cm}
\Def{The set $FV$ of free variables.}
    {A variable is $x$ considered \textit{free} in a lambda term if
     it refers to a binding outside the \textit{scope} of the 
     current lambda term. Defined inductively. \\
       \vspace{-0.5cm}
       \begin{align*}
         FV(V) &= V \\
         FV(M \; N) &= FV(M) \cup FV(N) \\
         FV(\lambda(V : \alpha).M) &= FV(M) \backslash V 
       \end{align*}
       \vspace{-0.5cm}
   }  
\Ex{Examples}
   { \vspace{-0.5cm}
     \begin{align*}
    \lambda(x : \alpha \rightarrow \beta).
      \lambda(y : \beta). x \; y \qquad &\text{$FV = \emptyset$} \\
    \lambda(x : \alpha \rightarrow \beta).
      \boxed{\lambda(y : \beta). x \; y} 
      \qquad &\text{FV = $\{x\}$} 
    \end{align*}}
}
